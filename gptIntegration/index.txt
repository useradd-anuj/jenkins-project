<script type="text/javascript">
	const products = [
		{
			id: 1,
			name: "Product 1",
			price: 10,
			image: "https://via.placeholder.com/400x400?text=Product+1"
		},
		{
			id: 2,
			name: "Product 2",
			price: 20,
			image: "https://via.placeholder.com/400x400?text=Product+2"
		},
		{
			id: 3,
			name: "Product 3",
			price: 30,
			image: "https://via.placeholder.com/400x400?text=Product+3"
		}
	];

	let cart = [];

	const cartItems = document.querySelector(".cart-items");

	const clearCartBtn = document.querySelector(".clear-cart");

	const addBtns = document.querySelectorAll(".add-to-cart");

	addBtns.forEach(addBtn => {
		addBtn.addEventListener("click", () => {
			const productId = parseInt(addBtn.dataset.productId);
			const product = products.find(product => product.id === productId);
			if (product) {
				let item = cart.find(item => item.id === productId);
				if (item) {
					item.quantity++;
				} else {
					item = {
						id: productId,
						name: product.name,
						price: product.price,
						image: product.image,
						quantity: 1
					};
					cart.push(item);
				}
				updateCart();
			}
		});
	});

	clearCartBtn.addEventListener("click", () => {
		cart = [];
		updateCart();
	});

	function updateCart() {
		cartItems.innerHTML = "";
		cart.forEach(item => {
			const itemEl = document.createElement("li");
			const nameEl = document.createElement("span");
			const priceEl = document.createElement("span");
			const quantityEl = document.createElement("span");
			const imageEl = document.createElement("img");
			const removeBtn = document.createElement("button");

			nameEl.textContent = item.name;
			priceEl.textContent = `$${item.price.toFixed(2)}`;
			quantityEl.textContent = `x${item.quantity}`;
			imageEl.src = item.image;
			removeBtn.textContent = "Remove";
			removeBtn.dataset.productId = item.id;

			nameEl.classList.add("item-name");
			priceEl.classList.add("item-price");
			quantityEl.classList.add("item-quantity");

			removeBtn.addEventListener("click", () => {
				const index = cart.findIndex(i => i.id === item.id);
				cart.splice(index, 1);
				updateCart();
			});

			itemEl.appendChild(nameEl);
			itemEl.appendChild(priceEl);
			itemEl.appendChild(quantityEl);
			itemEl.appendChild(removeBtn);
			itemEl.appendChild(imageEl);

			cartItems.appendChild(itemEl);
		});
	}

	updateCart();
</script>
